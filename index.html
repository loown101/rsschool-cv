  class Button {
    constructor(width, height, background, value) {
      this.width = +width;
      this.height = +height;
      this.background = background;
      this.value = value;
  }
            
    render() {
      let btn1 = document.createElement("button");
            
      btn1.style.width = this.width + 'px';
      btn1.style.height = this.height + 'px';
      btn1.style.background = this.background;
      btn1.innerHTML = this.value;
            
      return btn1;
    }
  }

  class ModernButton extends Button {
    constructor(width, height, background, value, borderRadius = 0) {
      super(width, height, background, value);
      this.borderRadius = +borderRadius;
  }
            
            
    render() {
     let btn2 = super.render();
            
      btn2.style.borderRadius = this.borderRadius + 'px';
            
      return btn2;
    }
  }

  let btn1 = new Button(125, 65, 'red', 'Привет')
  let btn2 = new ModernButton(150, 100, 'green', 'Нажми', 5)

  btn1.render();
  btn2.render();
